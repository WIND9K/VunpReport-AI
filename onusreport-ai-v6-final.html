🔧 BaseAgent — Interface chuẩn cho tất cả 5 agents
# agents/base_agent.py import asyncio from abc import ABC, abstractmethod class BaseAgent(ABC): def __init__(self, date, config): self.date = date self.config = config self.session_buffer = {} # in-memory per-run risk buffer @abstractmethod async def fetch(self) -> list[dict]: """Fetch raw data từ onuslibs — implement trong mỗi agent""" ... async def run(self): rows = await self.fetch() # onuslibs API call await asyncio.gather( self._etl_handler(rows), # → Aurora diary table self._parquet_handler(rows), # → /tmp/{group}_{date}.jsonl self._risk_handler(rows), # → risk_events nếu score ≥ threshold ) async def _etl_handler(self, rows): from etl.transform import enrich, normalize, aggregate data = aggregate(normalize(enrich(rows))) from core.db import bulk_upsert bulk_upsert(self.diary_table, data) async def _parquet_handler(self, rows): from datalake.parquet_writer import ParquetWriter writer = ParquetWriter() writer.append_batch(self.group_name, rows) async def _risk_handler(self, rows): # Process per batch, per userid — real-time risk in-memory for userid, user_rows in group_by(rows, "userid"): await self._process_batch(userid, user_rows) async def _process_batch(self, userid, user_rows): score, rules = risk_rules.check(userid, user_rows, self.session_buffer) if score >= THRESHOLD_MEDIUM: case_id = risk_events.create( userid=userid, agent_type=self.group_name, score=score, rules_triggered=rules, raw_evidence=user_rows, detection_layer="realtime" ) if score >= THRESHOLD_HIGH: await telegram.alert(case_id, userid, score, rules) elif score >= THRESHOLD_CRITICAL: await telegram.critical(case_id, userid, score, rules) await slack.alert(case_id, score, rules)

14ROLLBACK PLAN
Kế Hoạch Rollback — Khi Có Sự Cố
3 cấp độ rollback tương ứng với 3 mức nghiêm trọng
L1
LEVEL 1 — AGENT CRASH
🟡 1 Agent bị lỗi, các agent khác vẫn chạy
  • asyncio isolation: return_exceptions=True trong gather — fail 1 không dừng 4 còn lại
  • OpenClaw log + Telegram tự động: "⚠️ exchange_agent failed: {error_msg}"
  • finalize_skill bỏ qua group lỗi (kiểm tra temp file tồn tại trước khi convert)
  • Sáng: xem log → fix bug → chạy lại agent thủ công với data hôm qua
  • Data loss: 1 ngày của 1 agent. Diary table thiếu 1 ngày → backfill từ API Onus
L2
LEVEL 2 — TOÀN HỆ THỐNG LỖI
🔴 Tất cả agents crash, không có data
  • Ngay lập tức: onusreport cũ vẫn chạy độc lập → KPI báo cáo không bị gián đoạn
  • Xác định nguyên nhân: DB down? S3 outage? Server hết RAM? API Onus thay đổi schema?
  • APScheduler fallback đã kích hoạt chưa? Check log
  • Nếu API Onus thay đổi: update onuslibs → test → deploy trong 24h
  • Risk alerts gián đoạn: tăng manual review trong thời gian fix
L3
LEVEL 3 — DATA CORRUPTION
🟣 Diary tables bị ghi sai, risk_events không đúng
  • Stop tất cả agents ngay lập tức (kill cron jobs + APScheduler)
  • Restore Aurora từ snapshot (automated backup daily, giữ 7 ngày)
  • S3 Parquet append-only — không bị ảnh hưởng, dùng rebuild nếu cần
  • Replay data từ ngày bị lỗi: SHADOW_MODE=true, verify trước khi restore production
  • Post-mortem: cập nhật validation logic để ngăn tái diễn
🛡️
Nguyên tắc không bao giờ vi phạm
onusreport cũ vẫn chạy độc lập trong suốt quá trình deploy và vận hành onusreport-ai. Chỉ tắt onusreport cũ khi onusreport-ai đã chạy ổn định ≥30 ngày liên tiếp không có sự cố nghiêm trọng. Secret keys không bao giờ commit lên git. Shadow Mode 5 đêm không được skip.
